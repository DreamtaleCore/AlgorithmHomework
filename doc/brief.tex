  \subsection{Kruskal Algorithm}
      \subsubsection{Pseudo Code}
      \begin{algorithm}
      \caption{Kruskal algorithm}\label{kruskal}
      \begin{algorithmic}[1]
      \Function{Kruskal}{$V,E$}\Comment{$V$ denotes vertices, $E$ denotes edges}
        \Require A weighted, connected map which vertices set as $V$ and edges set as $E$.
        \Ensure Using map $G_{new}$ which describe the minimal spanning tree.
        \State $G_{new}\gets\left\{v_0, e_0 \mid v_0=V, e_0 \in \O  \right\}$\Comment{$v_0$ has the same vertices number as $V$, $e_0$ denotes empty set}
        \State $E_s\gets sortFromSmallToLarge\left(E\right)$
        \State $V_{connected}\gets \left\{v_0, v_1\mid\langle v_0, v_1\rangle \in E_s\left[0\right]\right\}$
        \ForAll{$e_i \in E_s$}\Comment{From small to large}
          \If{$\forall v_t \in V, v_t \in G_{new}$}
            \State break
          \EndIf
          \If{$v_0 \in V_{connect}\quad\mathrm{and}\quad v_1 \notin V_{connect}\quad\mathrm{s.t.}\quad\langle v_0, v_1\rangle \in e_i$}
            \State add $e_i$ to $G_{new}$
            \State add $v_1$ to $V_{connected}$
          \EndIf
        \EndFor
      \EndFunction
      \end{algorithmic}
      \end{algorithm}
